% Configuración de colores para elementos interactivos
\definecolor{linkcolor}{RGB}{0, 76, 153}  % Un azul más oscuro y elegante
\definecolor{citecolor}{RGB}{0, 76, 153}

% Configuración de hyperref para versiones digitales

  linkcolor=linkcolor, % Todos los links en azul (incluyendo TOC)
  urlcolor=linkcolor,  % URLs en azul
  citecolor=linkcolor, % Citas en azul

}
